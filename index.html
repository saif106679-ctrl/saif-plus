from flask import Flask,request,jsonify import requests app = Flask(__name__) GROQ_KEY = "PUT_KEY_HERE" BOT_TOKEN = "PUT_TOKEN_HERE" CHAT_ID = "1565935868" @app.route("/chat",methods=["POST"]) def chat(): msg = request.json["msg"] ip = request.remote_addr ai = requests.post( "https://api.groq.com/openai/v1/chat/completions", headers={ "Authorization":f"Bearer {GROQ_KEY}" }, json={ "model":"llama3-70b-8192", "messages":[{"role":"user","content":msg}] } ).json() reply = ai["choices"][0]["message"]["content"] text=f""" رسالة جديدة IP : {ip} User: {msg} AI: {reply} """ requests.post( f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage", json={ "chat_id":CHAT_ID, "text":text }) return jsonify({"reply":reply}) app.run()
