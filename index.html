async function sendMessage() { const text = userInput.value.trim(); if (text === "") return; // إظهار رسالة المستخدم في الموقع addMessage("user-msg", `root@grod:~# analyze --target ${text}`); // إظهار لوق التحليل (مثل الأفلام) const logDiv = addMessage("bot-msg", "> Initializing OSINT protocols...\n> Bypassing TikTok region locks...\n> Extracting metadata..."); try { const response = await fetch('/analyze', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({username: text}) }); const data = await response.json(); if(data.status === "success") { setTimeout(() => { typeEffect(logDiv, ` [SUCCESS] البيانات المستخرجة: ------------------------- الحساب: ${data.username} الدولة: ${data.country} المتابعين: ${data.followers} التقييم: ${data.quality} ------------------------- تم إنهاء العملية بنجاح.`); }, 1500); } else { typeEffect(logDiv, "[ERROR] فشل الوصول للحساب. تأكد من اليوزر."); } } catch (e) { addMessage("bot-msg", "[FATAL ERROR] Connection lost."); } userInput.value = ""; } function addMessage(type, text) { const div = document.createElement('div'); div.className = `message ${type}`; div.textContent = text; chatBox.appendChild(div); chatBox.scrollTop = chatBox.scrollHeight; return div; }
